<script>
document.addEventListener('DOMContentLoaded', () => {
  // 1. Sanitize-Funktion: entfernt Zeilenumbrüche und <br> in Einzeilern
  function sanitize(str) {
    return str
      .replace(/[\r\n]+/g, ' ')
      .replace(/<br\s*\/?>/gi, ' ');
  }

  // 2. Paste-Handler: fängt normale Paste-Ereignisse ab (ohne Clipboard-API-Prompt)
  document.addEventListener('paste', e => {
    const el = e.target;
    // 2.1 Nur contenteditable-Felder bearbeiten
    if (!el.isContentEditable) return;

    // 2.2 Default-Verhalten stoppen, um sauberen Text zu bekommen
    e.preventDefault();

    // 2.3 Reinen Text aus dem Event auslesen
    let text = e.clipboardData.getData('text/plain');

    // 2.4 Bei .oneline: Zeilenumbrüche & <br> entfernen
    if (el.classList.contains('oneline')) {
      text = sanitize(text);
    }

    // 2.5 Text an der Cursor-Position einfügen
    document.execCommand('insertText', false, text);
  }, true);

  // 3. Drop-Handler als Fallback (falls vorinput/drop nicht gefeuert wird)
  document.addEventListener('drop', e => {
    const el = e.target;
    if (!el.isContentEditable) return;
    e.preventDefault();

    let text = e.dataTransfer.getData('text/plain');
    if (el.classList.contains('oneline')) {
      text = sanitize(text);
    }
    document.execCommand('insertText', false, text);
  }, true);

  // 4. Einzeiler-spezifisch: Enter & Formatierungen blocken
  document.querySelectorAll('.oneline[contenteditable]').forEach(el => {
    // 4.1 Enter/Return verhindern
    el.addEventListener('keydown', e => {
      if (e.key === 'Enter' || e.key === 'Return') {
        e.preventDefault();
      }
    });
    // 4.2 Bestimmte Formatierungen blocken
    el.addEventListener('beforeinput', e => {
      const blocked = [
        'insertParagraph',
        'insertLineBreak',
        'formatBold',
        'formatItalic',
        'formatUnderline'
      ];
      if (blocked.includes(e.inputType)) {
        e.preventDefault();
      }
    });
  });

  // 5. Optionaler CSS-Hinweis für Einzeiler (besser per externem Stylesheet)
  // <style>
  //   .oneline[contenteditable] {
  //     white-space: nowrap;
  //     overflow: hidden;
  //   }
  // </style>
});
</script>




  
  
<script>
document.addEventListener('DOMContentLoaded', () => {
    // 1) Automatische <br>-Einfügung im Adressfeld
    const adresse = document.querySelector('adresse.schriftfeld[contenteditable]');
    if (adresse) {
        function formatAdresseBr() {
            let text = adresse.innerText.replace(/\s+/g, ' ').trim();
            // Regex: Straße/Hausnummer, 4/5-stellige PLZ, Ort
            const match = text.match(/^(.*?\d[\w\s\*\-]*)\s(\d{4,5})\s(.+)$/);
            if (match) {
                const street = match[1].trim();
                const plz = match[2];
                const city = match[3].trim();
                const htmlSoll = `${street}<br>${plz} ${city}`;
                if (adresse.innerHTML.replace(/<div>/g, '<br>').replace(/<\/div>/g, '') !== htmlSoll) {
                    adresse.innerHTML = htmlSoll;
                    // Cursor ans Ende setzen
                    setTimeout(() => {
                        const range = document.createRange();
                        const sel = window.getSelection();
                        range.selectNodeContents(adresse);
                        range.collapse(false);
                        sel.removeAllRanges();
                        sel.addRange(range);
                    }, 0);
                }
            }
        }
        adresse.addEventListener('input', formatAdresseBr);
        adresse.addEventListener('blur', formatAdresseBr);
    }

    // 2) Rücksendeangabe & Maschinenschriftzeile automatisch aktualisieren
    const absender = document.querySelector('absender');
    const rucksendeangabe = document.querySelector('rucksendeangabe');
    const maschunterschrift = document.querySelector('maschunterschrift');

    // Hilfsfunktion: ersten Buchstaben jedes Wortes groß
    function capitalizeWord(str) {
        return str.replace(/\S+/g, w => w.charAt(0).toUpperCase() + w.slice(1).toLowerCase());
    }

    // Name zu Initiale + Nachname
    function formatRucksendeName(name) {
        if (!name) return '';
        const trimmed = name.trim();
        const spaceIdx = trimmed.indexOf(' ');
        if (spaceIdx === -1) return capitalizeWord(trimmed);
        const initial = trimmed.charAt(0).toUpperCase() + '.';
        const nachname = capitalizeWord(trimmed.slice(spaceIdx + 1));
        return `${initial} ${nachname}`;
    }

    // Adresse: HTML (mit <br>) zu Fließtext mit " | " als Trenner
    function formatAdresseForRucksendeangabe(adresseEl) {
        if (!adresseEl) return '';
        // Ersetze <br> und Zeilenumbrüche durch " | "
        return adresseEl.innerHTML
            .replace(/<br\s*\/?>/gi, ' | ')
            .replace(/<div>/gi, ' | ')
            .replace(/<\/div>/gi, '')
            .replace(/[\r\n]+/g, ' ')
            .replace(/\s+/g, ' ')
            .trim();
    }

    function updateRucksendeangabeUndUnterschrift() {
        const absenderText = absender ? absender.innerText.trim() : '';
        const nameFormatted = formatRucksendeName(absenderText);
        const adresseText = adresse ? formatAdresseForRucksendeangabe(adresse) : '';
        let rucksendeText = nameFormatted;
        if (adresseText) {
            rucksendeText += ' | ' + adresseText;
        }
        if (rucksendeangabe) {
            rucksendeangabe.innerText = rucksendeText;
        }
        if (maschunterschrift) {
            maschunterschrift.innerText = absenderText.charAt(0).toUpperCase() + absenderText.slice(1);
        }
    }

    // Events für dynamische Aktualisierung
    if (absender) absender.addEventListener('input', updateRucksendeangabeUndUnterschrift);
    if (adresse) adresse.addEventListener('input', updateRucksendeangabeUndUnterschrift);

    // Initiales Setzen beim Laden
    updateRucksendeangabeUndUnterschrift();
});
</script>




<script>
document.addEventListener('DOMContentLoaded', () => {
  const telefon = document.querySelector('telefon.schriftfeld.oneline[contenteditable]');

  if (!telefon) return;

  // Listen für typische Vorwahlen (Mobil, Sondernummern, etc.)
  const mobilVorwahlen = [
    "0151","0152","0155","0156","0157","0159",
    "0160","0162","0163","0164","0165","0166","0167","0168","0169",
    "0170","0171","0172","0173","0174","0175","0176","0177","0178","0179"
  ];
  const sonderVorwahlen = [
    "0180","0181","0182","0183","0184","0185","0186","0187","0188","0189",
    "0800","0700","0900","032","030","040","050","060","070","080","090"
  ];

  // Hilfsfunktion: Findet die korrekte Vorwahl-Länge für deutsche Nummern
  function findeVorwahl(num) {
    // Mobilfunk
    for (const vwl of mobilVorwahlen) {
      if (num.startsWith(vwl)) return vwl.length;
    }
    // Sondernummern
    for (const vwl of sonderVorwahlen) {
      if (num.startsWith(vwl)) return vwl.length;
    }
    // Ortsnetz: 3-5-stellig nach 0 (z.B. 0221, 02241, 089)
    // Liste der längsten deutschen Vorwahlen (ohne Mobil/Sondernummern)
    // 5-stellig
    if (/^0\d{4}/.test(num)) return 5;
    // 4-stellig
    if (/^0\d{3}/.test(num)) return 4;
    // 3-stellig
    if (/^0\d{2}/.test(num)) return 3;
    return 0;
  }

  function formatPhoneDin5008(input) {
    // 1. Entferne unerlaubte Zeichen (Klammern, Schrägstriche, Punkte, Striche, Semikolon, Leerzeichen)
    let num = input.replace(/[\s\/\-\.\(\);]+/g, '');

    // 2. Prüfe auf internationale Nummer
    let intl = '';
    if (num.startsWith('+')) {
      intl = '+';
      num = num.slice(1);
    } else if (num.startsWith('00')) {
      intl = '+';
      num = num.slice(2);
    }

    // 3. Deutsche Nummer mit Landesvorwahl (49)
    if (intl && num.startsWith('49')) {
      num = num.slice(2);
      if (num.startsWith('0')) num = num.slice(1); // 0 nach 49 entfällt
      // Finde Mobil- oder Festnetzvorwahl
      let vwlLen = findeVorwahl(num);
      if (vwlLen < 3) vwlLen = 3; // Minimum für deutsche Vorwahlen
      const vorwahl = num.slice(0, vwlLen);
      let rest = num.slice(vwlLen);
      // Durchwahl erkennen
      let durchwahl = '';
      const dwMatch = rest.match(/(.+?)(\d{1,5})$/);
      if (/-/.test(input)) {
        // Durchwahl mit Bindestrich
        const [haupt, dw] = rest.split('-');
        rest = haupt;
        durchwahl = '-' + dw;
      }
      return `${intl}49 ${vorwahl} ${rest}${durchwahl}`;
    }

    // 4. Nationale Nummer (mit führender 0)
    if (num.startsWith('0')) {
      let vwlLen = findeVorwahl(num);
      if (vwlLen < 3) vwlLen = 3;
      const vorwahl = num.slice(0, vwlLen);
      let rest = num.slice(vwlLen);
      // Durchwahl erkennen
      let durchwahl = '';
      if (/-/.test(input)) {
        const [haupt, dw] = rest.split('-');
        rest = haupt;
        durchwahl = '-' + dw;
      }
      return `${vorwahl} ${rest}${durchwahl}`;
    }

    // 5. Internationale Nummern anderer Länder
    if (intl && !num.startsWith('49')) {
      const match = num.match(/^(\d{1,3})(\d+)$/);
      if (match) {
        let [, ländercode, rest] = match;
        return `${intl}${ländercode} ${rest}`;
      }
    }

    // 6. Fallback: Gib bereinigte Nummer zurück
    return input.replace(/[\s\/\-\.\(\);]+/g, '');
  }

  // Nur beim Verlassen des Feldes formatieren!
  telefon.addEventListener('blur', () => {
    const orig = telefon.innerText;
    const formatted = formatPhoneDin5008(orig);
    if (orig !== formatted) telefon.innerText = formatted;
  });
});
</script>
