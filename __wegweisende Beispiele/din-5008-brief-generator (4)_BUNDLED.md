- **Scope**: All `contenteditable` fields.

 * API Service for Address Autocomplete
 * NAMESPACE: window.App.API
 */

window.App = window.App || {};

window.App.API = {
    fetchAddressSuggestions: async function(query, config) {
        if (query.length < 3) return [];

        let url = '';
        
        if (config.provider === 'photon') {
            url = `https://photon.komoot.io/api/?q=${encodeURIComponent(query)}&lang=de&limit=5`;
        } else if (config.provider === 'geoapify' && config.apiKey) {
            url = `https://api.geoapify.com/v1/geocode/autocomplete?text=${encodeURIComponent(query)}&apiKey=${config.apiKey}&limit=5&lang=de`;
        } else {
            return [];
        }

        try {
            const res = await fetch(url);
            if (!res.ok) throw new Error('Network response was not ok');
            const data = await res.json();
            return this.parseFeatures(data.features, config.provider);
        } catch (err) {
            if (window.App.Logger) {
                window.App.Logger.error('API', `Fetch Error: ${err.message}`);
            } else {
                console.error("API Fetch Error:", err);
            }
            return [];
        }
    },

    parseFeatures: function(features, provider) {
        if (!features || !features.length) return [];

        return features.map(f => {
            const p = f.properties;
            let display = '';
            let insertion = '';

            if (provider === 'photon') {
                display = `${p.name || ''} ${p.street || ''} ${p.housenumber || ''}, ${p.city || ''}`;
                const name = p.name || '';
                const street = `${p.street || ''} ${p.housenumber || ''}`.trim();
                const city = `${p.postcode || ''} ${p.city || ''}`.trim();
                insertion = [name, street, city].filter(Boolean).join('\n');
            } else {
                // Geoapify
                display = p.formatted;
                insertion = `${p.address_line1}\n${p.address_line2}`;
            }
            
            return { display, insertion };
        });
    }
};

 * NAMESPACE: window.App.Storage

window.App.Storage = {
    STORAGE_KEY: 'din5008_data',

    saveLocal: function(data) {
        try {
            localStorage.setItem(this.STORAGE_KEY, JSON.stringify(data));
            return true;
        } catch (e) {
            if (window.App.Logger) window.App.Logger.error('Storage', 'Save failed: ' + e.message);
            return false;
        }
    },

    loadLocal: function() {
        const raw = localStorage.getItem(this.STORAGE_KEY);
        if (!raw) return null;
        try {
            return JSON.parse(raw);
        } catch (e) {
            if (window.App.Logger) window.App.Logger.error('Storage', 'Load failed: ' + e.message);
            return null;
        }
    },

    exportToJSON: function(data) {
        try {
            const blob = new Blob([JSON.stringify(data, null, 2)], {type: 'application/json'});
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `din5008_backup_${new Date().toISOString().slice(0,10)}.json`;
            a.click();
            URL.revokeObjectURL(url);
            if (window.App.Logger) window.App.Logger.log('Storage', 'Export generated successfully');
        } catch(e) {
            if (window.App.Logger) window.App.Logger.error('Storage', 'Export failed: ' + e.message);
        }
    },

    parseImportFile: function(file) {
        return new Promise((resolve, reject) => {
            const reader = new FileReader();
            reader.onload = (evt) => {
                try {
                    const data = JSON.parse(evt.target.result);
                    if (window.App.Logger) window.App.Logger.log('Storage', 'File parsed successfully');
                    resolve(data);
                } catch (err) {
                    if (window.App.Logger) window.App.Logger.error('Storage', 'JSON Parse Error: ' + err.message);
                    reject(err);
                }
            };
            reader.onerror = reject;
            reader.readAsText(file);
        });
    }
};

/**
 * Pure functions for string manipulation and formatting.
 * -------------------------------------------------------
 * SOURCE OF TRUTH: 03_javascript_knowledge.md
 * NAMESPACE: window.App.Formatter
 * -------------------------------------------------------
 */

window.App = window.App || {};

window.App.Formatter = {
    
    /**
     * Derives the single-line return address (Rücksendeangabe) from the full sender block.
     * 
     * @see [DIN-LOGIC-RET] Zone A Definition in 02_din5008_knowledge.md
     * @see 03_javascript_knowledge.md#2-data-automation-intelligence
     * 
     * @param {string} rawSenderText - The multi-line sender address.
     * @returns {string} The formatted single line (e.g. "M. Mustermann, Street 1, City")
     */
    deriveReturnLine: function(rawSenderText) {
        const text = rawSenderText.trim();
        if (!text) return '';

        const lines = text.split('\n').map(l => l.trim()).filter(l => l);
        if (lines.length === 0) return '';

        // 1. Process Name (Line 1)
        let formattedName = lines[0];
        const nameParts = formattedName.split(/\s+/);
        if (nameParts.length > 1) {
            const first = nameParts[0];
            const last = nameParts[nameParts.length - 1];
            if (!first.includes('.')) {
                formattedName = `${first.charAt(0)}. ${last}`;
            }
        }

        // 2. Concatenate Address
        const addressPart = lines.slice(1).join(', ');
        return addressPart ? `${formattedName}, ${addressPart}` : formattedName;
    },

    /**
     * Smart formatting for date input (auto-inserts dots).
     * Enforces DD.MM.YYYY structure.
     * 
     * @see [DIN-LOGIC-DATE] Date Formatting Rules
     * @param {string} input - Raw user input
     * @returns {string} Formatted string
     */
    formatAsDate: function(input) {
        // Strip non-digits
        let val = input.replace(/\D/g, '');
        
        // Auto-insert dots
        if (val.length > 2) val = val.slice(0, 2) + '.' + val.slice(2);
        if (val.length > 5) val = val.slice(0, 5) + '.' + val.slice(5);
        if (val.length > 10) val = val.slice(0, 10);
        
        return val;
    },

    /**
     * Returns a formatted date string for a Date object based on style.
     * @see [DIN-FIELD-DATE-PIMP-2026]
     * @param {Date} dateObj 
     * @param {string} format 'de' | 'iso' | 'long'
     * @returns {string}
     */
    getFormattedDate: function(dateObj, format) {
        if (!dateObj || isNaN(dateObj.getTime())) return '';

        const d = dateObj.getDate();
        const m = dateObj.getMonth() + 1;
        const y = dateObj.getFullYear();
        
        const dd = d < 10 ? '0' + d : d;
        const mm = m < 10 ? '0' + m : m;

        if (format === 'iso') {
            return `${y}-${mm}-${dd}`;
        } else if (format === 'long') {
            // German locale long format
            const months = ['Januar', 'Februar', 'März', 'April', 'Mai', 'Juni', 'Juli', 'August', 'September', 'Oktober', 'November', 'Dezember'];
            return `${d}. ${months[dateObj.getMonth()]} ${y}`;
        } else {
            // Default 'de'
            return `${dd}.${mm}.${y}`;
        }
    },

    /**
     * Validates if the string matches the selected Date format.
     * @see [DIN-FIELD-DATE-PIMP-2026]
     * @param {string} str 
     * @param {string} format 'de' | 'iso' | 'long'
     * @returns {boolean}
     */
    validateDate: function(str, format) {
        if (!str) return true; // Empty is valid (can be cleared)
        str = str.trim();

        if (format === 'iso') {
            // YYYY-MM-DD
            return /^\d{4}-\d{2}-\d{2}$/.test(str) && !isNaN(Date.parse(str));
        } else if (format === 'long') {
            // D. MMMM YYYY - lenient check on Month names
            // e.g. "1. Januar 2026"
            return /^\d{1,2}\.\s+[a-zA-ZäöüÄÖÜß]+\s+\d{4}$/.test(str);
        } else {
            // Default 'de': DD.MM.YYYY
            return /^\d{2}\.\d{2}\.\d{4}$/.test(str);
        }
    },

    /**
     * Helper to convert various string formats to ISO for the native picker.
     * Used when clicking the calendar icon to prepopulate the picker.
     */
    toISODate: function(input) {
        if (!input) return '';
        input = input.trim();

        // 1. Try DD.MM.YYYY
        if (/^\d{2}\.\d{2}\.\d{4}$/.test(input)) {
            const parts = input.split('.');
            return `${parts[2]}-${parts[1]}-${parts[0]}`;
        }

        // 2. Try YYYY-MM-DD
        if (/^\d{4}-\d{2}-\d{2}$/.test(input)) {
            return input;
        }

        // 3. Try Long Format (1. Januar 2024) -> Harder, but let's try basic parsing
        // We can just try Date.parse if the browser supports local formats, but likely it won't work well across browsers.
        // Implementation for "D. MMMM YYYY" manual parsing:
        const longMatch = input.match(/^(\d{1,2})\.\s+([a-zA-ZäöüÄÖÜß]+)\s+(\d{4})$/);
        if (longMatch) {
            const months = ['Januar', 'Februar', 'März', 'April', 'Mai', 'Juni', 'Juli', 'August', 'September', 'Oktober', 'November', 'Dezember'];
            const day = longMatch[1].padStart(2, '0');
            const monthName = longMatch[2];
            const year = longMatch[3];
            const monthIdx = months.findIndex(m => m.toLowerCase() === monthName.toLowerCase());
            if (monthIdx !== -1) {
                const mm = (monthIdx + 1).toString().padStart(2, '0');
                return `${year}-${mm}-${day}`;
            }
        }

        return '';
    },

    /**
     * Converts "YYYY-MM-DD" to "DD.MM.YYYY" (Legacy helper, kept for safety)
     */
    fromISODate: function(isoDate) {
        if (!isoDate) return '';
        const parts = isoDate.split('-');
        if (parts.length !== 3) return '';
        const [y, m, d] = parts;
        return `${d}.${m}.${y}`;
    },

    /**
     * Removes the word "Betreff:" from the subject line.
     * 
     * @see [DIN-LOGIC-SUBJ] Subject Line Constraint (No 'Betreff:' prefix)
     * @param {string} text 
     * @returns {string} Cleaned subject line
     */
    cleanSubject: function(text) {
        // Enforces DIN 5008 rule: No "Betreff:" prefix
        if (text.toLowerCase().includes('betreff:')) {
            return text.replace(/Betreff:/gi, '').trim();
        }
        return text;
    },

    /**
     * Validates Single-Line Constraints.
     * @see [DIN-FIELD-SINGLELINE-2026]
     * @param {string} text 
     * @param {number} maxChars 
     * @returns {Object} { valid: boolean, message: string }
     */
    validateSingleLine: function(text, maxChars = 100) {
        // Check for line breaks (Soft or Hard)
        if (text.includes('\n') || text.includes('\r')) {
            return { valid: false, message: 'Zeilenumbrüche sind in diesem Feld nicht erlaubt.' };
        }
        
        if (text.length > maxChars) {
            return { valid: false, message: `Text zu lang (Max: ${maxChars} Zeichen).` };
        }
        
        return { valid: true };
    },

    /**
     * Analyzes the recipient block to extract gender and surname.
     * @see [DIN-LOGIC-GREETING-2026]
     * @param {string} text 
     * @returns {Object} { gender: 'm'|'f'|'n', name: string }
     */
    parseRecipient: function(text) {
        const lines = text.split('\n');
        let gender = 'n';
        let name = '';

        for (const line of lines) {
            const clean = line.trim();
            
            // Check for Male pattern
            if (/(^|\s)(Herr|Herrn)(\s|$)/i.test(clean)) {
                gender = 'm';
                // Extract Name: "Herrn Max Mustermann" -> "Mustermann"
                // Or "Herrn Dr. Müller" -> "Dr. Müller"
                const parts = clean.split(/\s+/);
                // Simple heuristic: Take the last word as surname if length > 1
                if (parts.length > 1) name = parts[parts.length - 1];
                // Better heuristic needed? For now, last word is mostly safe for surname
                // If full line is "Herrn Max Mustermann", we want "Mustermann". 
                // But for "Hallo..." we might want "Max". 
                // Let's stick to surname for formal/polite.
                break;
            }
            
            // Check for Female pattern
            if (/(^|\s)(Frau)(\s|$)/i.test(clean)) {
                gender = 'f';
                const parts = clean.split(/\s+/);
                if (parts.length > 1) name = parts[parts.length - 1];
                break;
            }
        }

        return { gender, name };
    },

    /**
     * Generates a salutation based on analysis and formality.
     * @see [DIN-LOGIC-GREETING-2026]
     */
    deriveSalutation: function(analysis, formality) {
        const { gender, name } = analysis;

        // Default Fallback
        if (gender === 'n' || !name) {
            return 'Sehr geehrte Damen und Herren,';
        }

        if (formality === 'casual') {
            return `Hallo Herr ${name},`; // Simplistic, ideally first name for casual
        } else if (formality === 'polite') {
            if (gender === 'm') return `Guten Tag Herr ${name},`;
            if (gender === 'f') return `Guten Tag Frau ${name},`;
        } else {
            // Formal
            if (gender === 'm') return `Sehr geehrter Herr ${name},`;
            if (gender === 'f') return `Sehr geehrte Frau ${name},`;
        }
        
        return 'Sehr geehrte Damen und Herren,';
    },

    /**
     * Generates a closing phrase based on formality.
     * @see [DIN-LOGIC-GREETING-2026]
     */
    deriveClosing: function(formality) {
        switch (formality) {
            case 'casual': return 'Viele Grüße';
            case 'polite': return 'Beste Grüße';
            case 'formal': 
            default: return 'Mit freundlichen Grüßen';
        }
    }
};

/**
 * Debug Logger Logic (Black Box Edition)
 * -------------------------------------------------------
 * RESPONSIBILITY: Global Error Trapping, Visual Console, Persistence.
 * NAMESPACE: window.App.Logger
 * FEATURE: Saves last 50 logs to LocalStorage for crash analysis.
 * -------------------------------------------------------
 */

window.App = window.App || {};

window.App.Logger = {
    el: null,
    contentEl: null,
    isCollapsed: true,
    history: [], // RAM Buffer
    MAX_HISTORY: 50,
    STORAGE_KEY: 'din_crash_log',

    init: function() {
        this.loadBlackBox();
        this.render();
        this.attachGlobalHandlers();
        
        this.checkPreviousCrash();
        this.log('System', 'Logger initialized (Black Box Active).');
    },

    /**
     * Load previous session logs from storage
     */
    loadBlackBox: function() {
        try {
            const raw = localStorage.getItem(this.STORAGE_KEY);
            if (raw) {
                this.history = JSON.parse(raw);
            }
        } catch (e) {
            console.error('Failed to load crash log', e);
            this.history = [];
        }
    },

    /**
     * Check if the last session ended with an error
     */
    checkPreviousCrash: function() {
        if (this.history.length > 0) {
            const lastEntry = this.history[this.history.length - 1];
            if (lastEntry.level === 'error') {
                // Delay slightly to ensure UI is ready
                setTimeout(() => {
                    this.error('CRASH_DETECTED', `Previous session ended with error: ${lastEntry.message}`);
                    this.toggle(false); // Auto-open console
                }, 500);
            }
        }
    },

    render: function() {
        // Create container if not exists in HTML
        let container = document.getElementById('debug-console');
        if (!container) {
            container = document.createElement('div');
            container.id = 'debug-console';
            container.className = 'no-print collapsed';
            document.body.appendChild(container);
        }

        // Structure
        container.innerHTML = `
            <div class="debug-header">
                <span class="debug-title">🔧 Flight Recorder</span>
                <div class="debug-controls">
                    <button id="btn-export-log" title="Download Log">💾</button>
                    <button id="btn-clear-log" title="Clear Console">🚫</button>
                    <button id="btn-toggle-log">▲</button>
                </div>
            </div>
            <div class="debug-content" id="debug-content"></div>
        `;

        this.el = container;
        this.contentEl = document.getElementById('debug-content');

        // Bind Controls
        document.getElementById('btn-toggle-log').addEventListener('click', () => this.toggle());
        document.getElementById('btn-clear-log').addEventListener('click', () => this.clear());
        document.getElementById('btn-export-log').addEventListener('click', () => this.exportLog());
    },

    toggle: function(forceState) {
        if (typeof forceState === 'boolean') {
            this.isCollapsed = forceState;
        } else {
            this.isCollapsed = !this.isCollapsed;
        }
        this.el.classList.toggle('collapsed', this.isCollapsed);
        document.getElementById('btn-toggle-log').textContent = this.isCollapsed ? '▲' : '▼';
    },

    clear: function() {
        this.contentEl.innerHTML = '';
        // Note: We do NOT clear the history/storage here, only the view, 
        // to preserve black box data for developers.
        this.log('System', 'Console view cleared (History preserved).');
    },

    /**
     * Exports the black box log as a text file
     */
    exportLog: function() {
        const textContent = this.history.map(entry => 
            `[${entry.time}] [${entry.level.toUpperCase()}] [${entry.context}] ${entry.message}`
        ).join('\n');

        const blob = new Blob([textContent], {type: 'text/plain'});
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = `din_crash_log_${new Date().toISOString().slice(0,19).replace(/:/g, '-')}.txt`;
        a.click();
        URL.revokeObjectURL(url);
    },

    /**
     * Persist to LocalStorage (FIFO)
     */
    persist: function(entry) {
        this.history.push(entry);
        if (this.history.length > this.MAX_HISTORY) {
            this.history.shift();
        }
        try {
            localStorage.setItem(this.STORAGE_KEY, JSON.stringify(this.history));
        } catch (e) {
            // Storage likely full, ignore to prevent recursive errors
        }
    },

    /**
     * Internal generic printer
     */
    print: function(level, context, message) {
        const timestamp = new Date().toLocaleTimeString();
        
        // 1. Update View
        if (this.contentEl) {
            const entryDiv = document.createElement('div');
            entryDiv.className = `log-entry log-${level}`;
            entryDiv.innerHTML = `
                <span class="log-time">[${timestamp}]</span>
                <span class="log-context">${context}:</span>
                <span class="log-msg">${message}</span>
            `;
            this.contentEl.appendChild(entryDiv);
            this.contentEl.scrollTop = this.contentEl.scrollHeight;
        }

        // 2. Persist
        this.persist({
            time: timestamp,
            level: level,
            context: context,
            message: message
        });

        // 3. Auto-expand on error
        if (level === 'error' && this.isCollapsed) {
            this.toggle(false);
        }
    },

    log: function(context, message) {
        console.log(`[${context}] ${message}`);
        this.print('info', context, message);
    },

    warn: function(context, message) {
        console.warn(`[${context}] ${message}`);
        this.print('warn', context, message);
    },

    error: function(context, message) {
        console.error(`[${context}] ${message}`);
        this.print('error', context, message);
    },

    attachGlobalHandlers: function() {
        window.onerror = (msg, url, lineNo, columnNo, error) => {
            this.error('GLOBAL_TRAP', `${msg} (Line: ${lineNo})`);
            return false;
        };
        
        window.onunhandledrejection = (event) => {
            this.error('PROMISE_TRAP', `Unhandled Rejection: ${event.reason}`);
        };
    }
};
